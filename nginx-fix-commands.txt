# Komendy do naprawy nginx - wykonaj po kolei:

# 1. Pobierz najnowsze zmiany z git
git pull origin main

# 2. Sprawdź czy plik konfiguracyjny istnieje
ls -la nginx-losuje-production.conf

# 3. Skopiuj konfigurację nginx
cp nginx-losuje-production.conf /etc/nginx/sites-available/losuje.pl

# 4. Aktywuj stronę
ln -sf /etc/nginx/sites-available/losuje.pl /etc/nginx/sites-enabled/

# 5. Sprawdź konfigurację nginx
nginx -t

# 6. Przeładuj nginx
systemctl reload nginx

# 7. Sprawdź status
systemctl status nginx

# 8. Sprawdź domeny
curl -I https://losuje.pl
curl -I https://losuje-generator.pl

# 9. Sprawdź backend
curl http://localhost:3001/health
